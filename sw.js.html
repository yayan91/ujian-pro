const CACHE_NAME = 'mappaguru-cbt-v2'; // Versi Cache (Ubah angka jika ada update besar) // Daftar file yang WAJIB ada di GitHub dengan nama yang SAMA PERSIS hurufnya const urlsToCache = [ './', './index.html', './manifest.json', './logo-192.png', './logo-512.png' ]; // PROSES INSTALL & CACHING self.addEventListener('install', event => { self.skipWaiting(); // Paksa agen rahasia baru langsung aktif bekerja event.waitUntil( caches.open(CACHE_NAME) .then(cache => { console.log('Membuka cache Mappaguru...'); return cache.addAll(urlsToCache); }) .catch(err => console.log('Ada file yang gagal di-cache (cek nama file!):', err)) ); }); // PROSES AKTIVASI & BERSIH-BERSIH CACHE LAMA self.addEventListener('activate', event => { event.waitUntil( caches.keys().then(cacheNames => { return Promise.all( cacheNames.filter(cacheName => cacheName !== CACHE_NAME) .map(cacheName => caches.delete(cacheName)) ); }) ); }); // PROSES PENGAMBILAN DATA (FETCH) self.addEventListener('fetch', event => { // Biarkan jalur ke Google Apps Script tetap jalan pakai internet if (event.request.url.includes('script.google.com')) { return; } // Cek apakah file ada di memori HP, kalau ada tampilkan, kalau tidak ambil dari internet event.respondWith( caches.match(event.request) .then(response => { return response || fetch(event.request); }) ); });
