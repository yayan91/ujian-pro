const CACHE_NAME = 'mappaguru-exam-v3'; // Daftar file yang WAJIB ada di GitHub dengan nama yang persis sama const urlsToCache = [ './', './index.html', './manifest.json', './1234.png', './909090.png' ]; self.addEventListener('install', event => { self.skipWaiting(); event.waitUntil( caches.open(CACHE_NAME) .then(cache => { console.log('Membuka cache Mappaguru...'); return cache.addAll(urlsToCache); }) .catch(err => console.error('Gagal caching file:', err)) ); }); self.addEventListener('activate', event => { event.waitUntil( caches.keys().then(cacheNames => { return Promise.all( cacheNames.filter(cacheName => cacheName !== CACHE_NAME) .map(cacheName => caches.delete(cacheName)) ); }) ); }); self.addEventListener('fetch', event => { // Biarkan jalur Google Apps Script tetap live if (event.request.url.includes('script.google.com')) { return; } event.respondWith( caches.match(event.request) .then(response => { return response || fetch(event.request); }) ); });
